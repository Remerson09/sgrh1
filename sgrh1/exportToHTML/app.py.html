from flask import Flask, render_template, request

class Profissao:
    def __init__(self, cod_profissao, nome, status):
        self.cod_profissao = cod_profissao
        self.nome = nome
        self.status = status

class Pessoa:
    def __init__(self, nome, cpf, endereco, profissao, senha):
        self.nome = nome
        self.cpf = cpf
        self.endereco = endereco
        self.profissao = profissao
        self.senha = senha

pessoa0 = Pessoa("Alice Silva", "123.456.789-00", "Rua das Flores, 123", "Engenheira", "123")
pessoa1 = Pessoa("Bruno Santos", "987.654.321-00", "Avenida Brasil, 456", "Médico", "123")
pessoa2 = Pessoa("Carla Oliveira", "234.567.890-11", "Travessa do Sol, 789", "Professora", "123")
pessoa3 = Pessoa("Daniela Costa", "345.678.901-22", "Rua da Paz, 101", "Arquiteta", "123")
lista = [pessoa0, pessoa1, pessoa2, pessoa3]
profissao0 = Profissao(0, "Engenheiro", "Disponível")
listap = [profissao0]

app = Flask(__name__)

@app.route('/index')
def ola():
    return render_template('menu.html', titulo="Sistemas de Gestão de Recursos Humanos")

@app.route('/form', methods=['GET', 'POST'])
def form_pessoa():
    if request.method == 'POST':
        nome = request.form['nome']
        cpf = request.form['cpf']
        endereco = request.form['endereco']
        profissao = request.form['profissao']
        return render_template('index.html', titulo='Dados Recebidos', nome=nome, cpf=cpf, endereco=endereco, profissao=profissao)

    return render_template('form.html', titulo="Sistemas de Gestão de Recursos Humanos")

@app.route('/form_profissao', methods=['GET', 'POST'])
def form_profissao():
    if request.method == 'POST':
        nome = request.form['nome']
        status = request.form['status']
        return render_template('index.html', titulo='Dados Recebidos', nome=nome, status=status)

    return render_template('form_profissao.html', titulo="Sistemas de Gestão de Recursos Humanos")

@app.route('/criar', methods=['GET', 'POST'])
def criar():
    if request.method == 'POST':
        nome = request.form['nome']
        cpf = request.form['cpf']
        endereco = request.form['endereco']
        profissao = request.form['profissao']
        pessoaR = Pessoa(nome, cpf, endereco, profissao)
        lista.append(pessoaR)
        return render_template('lista.html', titulo="Sistemas de Gestão de Recursos Humanos", pessoas=lista)

@app.route('/criar_profissao', methods=['GET', 'POST'])
def criar_profissao():
    if request.method == 'POST':
        nome = request.form['nome']
        status = request.form['status']
        profissaoR = Profissao(len(listap),nome, status)
        listap.append(profissaoR)
        return render_template('saida_profissao.html', titulo="Sistemas de Gestão de Recursos Humanos", profissaos=listap)

if __name__ == '__main__':
    app.run(debug=True)

@app.route('/login')
def login():
    return render_template("login.html", titulo="Sistemas de Gestão de Recursos Humanos")

@app.route('/autenticar', methods=['GET', 'POST'])
def autenticar():
    if request.method == 'POST':
        user = request.form['user']
        for pessoa in lista:
            if pessoa.nome == user and pessoa.senha == request.form['password']:
                return render_template('lista.html', titulo="Sistemas de Gestão de Recursos Humanos", pessoas=lista)
            else:
                return render_template('login.html', titulo="Sistemas de Gestão de Recursos Humanos")
